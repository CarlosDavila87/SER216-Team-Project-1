	void addGamePanel(){
		gamePanel = new GamePanel(this, mainMenu.getIsEnabled());
		add(gamePanel);
	}
	
	void removeGamePanel(){
		remove(gamePanel);
	}
	
	void addMainMenu(){
		score1 = 0;
		score2 = 0;
		mainMenu = new MainMenuPanel(this);
		add(mainMenu);
	}

		System.out.println("New Game Over Panel added");
		gameOverPanel = new GameOverPanel(this, winner);
		add(gameOverPanel);
		updateDisplay();
	}
